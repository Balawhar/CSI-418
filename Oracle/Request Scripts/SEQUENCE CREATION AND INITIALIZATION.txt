   CREATE SEQUENCE VLPDBA.S_NAME MINVALUE 0 MAXVALUE 9999999999 START WITH 1 INCREMENT BY 1 CACHE 20 ORDER;
CREATE OR REPLACE PUBLIC SYNONYM S_NAME FOR OWNER.S_NAME;



INSERT INTO SDEDBA.REF_SYS_TABLE_SEQUENCE
  (OWNER, TABLE_NAME, COLUMN_NAME, SEQUENCE_NAME)
  SELECT 'OWNER',
         'TABLE_NAME',
         'COLUMN',
         'S_NAME'
    FROM DUAL
   WHERE ('TABLE_NAME', 'S_NAME') NOT IN
         (SELECT TABLE_NAME, SEQUENCE_NAME
            FROM SDEDBA.REF_SYS_TABLE_SEQUENCE);
COMMIT;

Exec SUITEDBA.STRUCTURE_DIAGNOSTIC.SEQUENCE_INITIALIZATION('OWNER','TABLE_NAME');
