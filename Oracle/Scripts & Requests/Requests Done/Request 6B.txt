
/*

OWNER	TABLE	COLUMN	DATA TYPE	DATA NULL	default value
HRDBA	HR_PERSON	MONTHLY_INCOME	NUMBER	y	 

*/


DECLARE 
NFOUND NUMBER;
BEGIN
    SELECT COUNT(*) INTO NFOUND FROM ALL_TAB_COLUMNS 
    WHERE TABLE_NAME = 'HR_PERSON' 
    AND OWNER = 'HRDBA' 
    AND COLUMN_NAME = 'MONTHLY_INCOME';
    IF NFOUND = 0 THEN
    EXECUTE IMMEDIATE 'ALTER TABLE HRDBA.HR_PERSON ADD MONTHLY_INCOME NUMBER';
    END IF;
END;
/
