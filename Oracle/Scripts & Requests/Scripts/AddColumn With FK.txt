
DECLARE 
NFOUND NUMBER;
BEGIN
    SELECT COUNT(*) INTO NFOUND FROM ALL_TAB_COLUMNS 
    WHERE TABLE_NAME = 'HR_PERSON' 
    AND OWNER = 'HRDBA' 
    AND COLUMN_NAME = 'CUSTOMER_ID';
    IF NFOUND = 0 THEN
    EXECUTE IMMEDIATE 'ALTER TABLE HRDBA.HR_PERSON ADD CUSTOMER_ID NUMBER(10)';
    EXECUTE IMMEDIATE 'ALTER TABLE HRDBA.HR_PERSON ADD CONSTRAINT "FK05_HR_PERSON" FOREIGN KEY (CUSTOMER_ID)
        REFERENCES SDEDBA.REF_CUSTOMER(CUSTOMER_ID) ENABLE';
    EXECUTE IMMEDIATE 'CREATE INDEX HRDBA.IDX05_HR_PERSON ON HRDBA.HR_PERSON(CUSTOMER_ID) TABLESPACE KERNEL_INDEX';
    END IF;
END;
/
